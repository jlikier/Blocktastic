    Vector3F position
    Vector3F velocity
    int durability //how many hits left before breaking
    int lastHitSeconds //how many seconds ago was it hit
    
blocklib.map.block.BlockType
    int typeID
    Vector3F dimension
    boolean breakable
    int maxDurability //how many hits does it take to break
    int resetDurabilitySeconds //how many seconds does it take to reset durability
    int textureID //which texture does this block map to
    float friction
    boolean collidable
    boolean transparent

    int targettingID //the uniqueid that this entity is targetting
    int textureID //the current texture
    Vector3F position
    Vector3F orientation
    Vector3F dimension
    Vector3F velocity
    int health
    int mana
    int level
    int state //bitwise-or to store values
